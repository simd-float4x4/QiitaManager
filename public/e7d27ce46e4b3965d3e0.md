# コード
* Extension.swiftをディレクトリの中にぶちこめばどこからでもアクセスできます。
```extension.swift
extension Date {
    // "."
    func toDotString() -> String {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "ja_JP")
        formatter.timeZone = TimeZone(identifier: "Asia/Tokyo")
        formatter.dateFormat = "yyyy.MM.dd"
        return formatter.string(from: self)
    }
    
    // "/"
     func toSlashString() -> String {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "ja_JP")
        formatter.timeZone = TimeZone(identifier: "Asia/Tokyo")
        formatter.dateFormat = "yyyy/MM/dd"
        return formatter.string(from: self)
    }

    // "漢字"
     func toKanjiString() -> String {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "ja_JP")
        formatter.timeZone = TimeZone(identifier: "Asia/Tokyo")
        formatter.dateFormat = "yyyy年MM月dd日"
        return formatter.string(from: self)
    }

    // "クリップ"
     func toClipString() -> String {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "ja_JP")
        formatter.timeZone = TimeZone(identifier: "Asia/Tokyo")
        formatter.dateFormat = "yyyyMMdd"
        return formatter.string(from: self)
    }

    // "カスタム"
     func toCustomString(separator: String) -> String {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "ja_JP")
        formatter.timeZone = TimeZone(identifier: "Asia/Tokyo")
        formatter.dateFormat = "yyyy\(separator)MM\(separator)dd"
        return formatter.string(from: self)
    }
}

```

# 呼び出し例
```ParentView.Swift
 VStack(spacing: 16) {
     // "2025.11.01"
    Text(item.created_at.toDotString()) 
        .font(.body)
        
    // "2025/11/02"
    Text(item.created_at.toSlashString()) 
        .font(.body)
        
    // "2025年11月03日"
    Text(item.created_at.toKanjiString()) 
        .font(.body)

    // "20251104"
    Text(item.created_at.toClipString()) 
        .font(.body)

    // "2025&11&05"
    Text(item.created_at.toCustomString(separator: "&")) 
        .font(.body)
}
```
